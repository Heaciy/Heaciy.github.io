import numpy
import matplotlib.pyplot
import scipy.special
%matplotlib inline

class neuralNetwork:
    def __init__(self, inputnodes, hiddennodes, outputnodes, learningrate):  # ÂàùÂßãÂåñÁ•ûÁªèÁΩëÁªú
        self.inodes = inputnodes
        self.hnodes = hiddennodes
        self.onodes = outputnodes
        self.lr = learningrate  # Â≠¶‰π†Áéá
        # from node i to node j in the next layer
        self.wih = numpy.random.normal(0.0, pow(self.hnodes, -0.5), (self.hnodes, self.inodes))  # ÊùÉÈáçÁü©Èòµ
        self.who = numpy.random.normal(0.0, pow(self.onodes, -0.5), (self.onodes, self.hnodes))
        self.activation_function = lambda x: scipy.special.expit(x)  # Â∫îÁî®ÊøÄÊ¥ªÂáΩÊï∞

    def train(self, inputs_list, targets_list):
        inputs = numpy.array(inputs_list, ndmin=2).T
        targets = numpy.array(targets_list, ndmin=2).T
        hidden_inputs = numpy.dot(self.wih, inputs)
        hidden_outputs = self.activation_function(hidden_inputs)
        final_inputs = numpy.dot(self.who, hidden_outputs)
        final_outputs = self.activation_function(final_inputs)

        output_errors = targets - final_outputs
        hidden_errors = numpy.dot(self.who.T, output_errors)

        self.who += self.lr*numpy.dot((output_errors*final_outputs*(1.0-final_outputs)), numpy.transpose(hidden_outputs))  # Ë∞ÉÊï¥ÈöêËóèÂ±Ç‰∏éËæìÂá∫Â±Ç‰πãÈó¥ÁöÑÊùÉÈáç
        self.wih += self.lr*numpy.dot((hidden_errors*hidden_outputs*(1.0-hidden_outputs)), numpy.transpose(inputs))  # Ë∞ÉÊï¥ËæìÂÖ•Â±Ç‰∏éÈöêËóèÂ±Ç‰πãÈó¥ÁöÑÊùÉÈáç

    def query(self, inputs_list):
        inputs = numpy.array(inputs_list, ndmin=2).T  # ËæìÂÖ•Áü©ÈòµÁöÑËΩ¨ÁΩÆÁü©Èòµ
        hidden_inputs = numpy.dot(self.wih, inputs)
        hidden_outputs = self.activation_function(hidden_inputs)
        final_inputs = numpy.dot(self.who, hidden_outputs)
        final_outputs = self.activation_function(final_inputs)
        return final_outputs

input_nodes = 784  # 28*28ÔºåËæìÂÖ•Â±ÇËäÇÁÇπÊï∞
hidden_nodes = 100  # ÈöêËóèÂ±ÇËäÇÁÇπÊï∞
output_nodes = 10  # ËæìÂá∫Â±ÇËäÇÁÇπ

learning_rate = 0.2  # Â≠¶‰π†Áéá

# ÁîüÊàê‰∏Ä‰∏™Á•ûÁªèÁΩëÁªúÂÆû‰æã
n = neuralNetwork(input_nodes,hidden_nodes,output_nodes,learning_rate)

# Âä†ËΩΩËÆ≠ÁªÉÊï∞ÊçÆ
training_data_file = open("mnist_dataset/mnist_train.csv", 'r')
training_data_list = training_data_file.readlines()
training_data_file.close()

# ËÆ≠ÁªÉÁ•ûÁªèÁΩëÁªú
epochs = 5  # ËÆ≠ÁªÉ5‰∏™‰∏ñ‰ª£(Êú¨ÁØáÁöÑÂ≠¶‰π†Áéá‰∏∫0.2ÔºåËÆ≠ÁªÉÊï∞ÊçÆ‰∏∫ÂÖ≠‰∏áÁªÑÔºåËÆ≠ÁªÉ‰∫î‰∏™‰∏ñ‰ª£ÔºåÊµãËØïÊï∞ÊçÆ‰∏∫‰∏Ä‰∏áÁªÑ)
for e in range(epochs):
    for record in training_data_list:
        all_values = record.split(',')
        inputs = (numpy.asfarray(all_values[1:])/255.0*0.99)+0.01
        targets = numpy.zeros(output_nodes) + 0.01
        targets[int(all_values[0])] = 0.99
        n.train(inputs,targets)

# Âä†ËΩΩÊµãËØïÊï∞ÊçÆ(Êú¨ÁØá‰ΩøÁî®‰∫Ü‰∏Ä‰∏áÁªÑÊµãËØïÊï∞ÊçÆÔºåÊ≠£Á°ÆÁéá‰∏∫95.11%ÔºåÁªìÊûúÂèØËÉΩ‰∏çÂ§ß‰∏ÄËá¥Ôºå‰ΩÜÂ§ßÊ¶ÇÈÉΩÂú®94%~95%)

    if(label == correct_label):
